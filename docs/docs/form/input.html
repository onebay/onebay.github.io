import { defineComponent, ref } from 'vue'

import { Input, Form, Toast } from 'onebay-ui'

export default defineComponent({
  name: 'InputPage',
  components: {
    Input,
    Form
  },
  data() {
    return {

      disabled: false,
      second: 60
    }
  },
  methods: {
    setState(state) {
      Object.keys(state).forEach((k) => {
        this[k] = state[k]
      })
    },
    showTipText() {
      return this.disabled ? `${this.second}s` : ''
    },

    sendCode() {
      if (this.disabled) return
      this.setState({
        disabled: true
      })
      // 
      const timer = setInterval(() => {
        if (this.second > 0) {
          this.setState({
            second: this.second - 1
          })
        } else {
          this.setState({
            second: 60,
            disabled: false
          })
          clearInterval(timer)
        }
      }, 1000)
    },

    handleInput(stateName, value) {
      this.setState({
        [stateName]: value
      })
      console.log(stateName, value)
    },

    onClickErrorIcon() {
      Toast({
        text: '',
        status: 'error',
        duration: 2000
      })
    }
  }
})
</script>
